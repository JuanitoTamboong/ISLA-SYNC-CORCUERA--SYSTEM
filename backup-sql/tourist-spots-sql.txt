-- ============================================
-- ISLA-SYNC CORCUERA - TOURIST SPOTS & SOUVENIRS TABLES
-- ============================================

-- ============================================
-- TOURIST SPOTS TABLE
-- ============================================

-- Create the tourist_spots table
CREATE TABLE IF NOT EXISTS public.tourist_spots (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    name TEXT NOT NULL,
    category TEXT NOT NULL CHECK (category IN ('Beach', 'Landmark')),
    description TEXT,
    location TEXT,
    image_url TEXT,
    is_visible BOOLEAN DEFAULT true,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Create indexes for faster queries
CREATE INDEX IF NOT EXISTS idx_tourist_spots_category ON public.tourist_spots(category);
CREATE INDEX IF NOT EXISTS idx_tourist_spots_is_visible ON public.tourist_spots(is_visible);
CREATE INDEX IF NOT EXISTS idx_tourist_spots_created_at ON public.tourist_spots(created_at);

-- Create trigger for updated_at
DROP TRIGGER IF EXISTS update_tourist_spots_updated_at ON public.tourist_spots;
CREATE TRIGGER update_tourist_spots_updated_at
    BEFORE UPDATE ON public.tourist_spots
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();

-- Enable Row Level Security
ALTER TABLE public.tourist_spots ENABLE ROW LEVEL SECURITY;

-- Policy 1: Admins can manage all tourist spots (CRUD)
CREATE POLICY "admins_manage_tourist_spots"
    ON public.tourist_spots
    FOR ALL
    USING (check_is_admin(auth.uid()))
    WITH CHECK (check_is_admin(auth.uid()));

-- Policy 2: Everyone can view visible tourist spots
CREATE POLICY "public_view_tourist_spots"
    ON public.tourist_spots
    FOR SELECT
    TO anon, authenticated
    USING (is_visible = true);

-- Grant permissions
GRANT ALL ON public.tourist_spots TO authenticated;
GRANT SELECT ON public.tourist_spots TO anon;
GRANT ALL ON public.tourist_spots TO service_role;


-- ============================================
-- SOUVENIRS TABLE
-- ============================================

-- Create the souvenirs table
CREATE TABLE IF NOT EXISTS public.souvenirs (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    tourist_spot_id UUID NOT NULL,
    name TEXT NOT NULL,
    description TEXT,
    price NUMERIC(10,2) NOT NULL CHECK (price >= 0),
    image_url TEXT,
    is_available BOOLEAN DEFAULT true,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    FOREIGN KEY (tourist_spot_id) REFERENCES public.tourist_spots(id) ON DELETE CASCADE
);

-- Create indexes for faster queries
CREATE INDEX IF NOT EXISTS idx_souvenirs_tourist_spot ON public.souvenirs(tourist_spot_id);
CREATE INDEX IF NOT EXISTS idx_souvenirs_is_available ON public.souvenirs(is_available);
CREATE INDEX IF NOT EXISTS idx_souvenirs_price ON public.souvenirs(price);
CREATE INDEX IF NOT EXISTS idx_souvenirs_created_at ON public.souvenirs(created_at);

-- Create trigger for updated_at
DROP TRIGGER IF EXISTS update_souvenirs_updated_at ON public.souvenirs;
CREATE TRIGGER update_souvenirs_updated_at
    BEFORE UPDATE ON public.souvenirs
    FOR EACH ROW
    EXECUTE FUNCTION update_updated_at_column();

-- Enable Row Level Security
ALTER TABLE public.souvenirs ENABLE ROW LEVEL SECURITY;

-- Policy 1: Admins can manage all souvenirs (CRUD)
CREATE POLICY "admins_manage_souvenirs"
    ON public.souvenirs
    FOR ALL
    USING (check_is_admin(auth.uid()))
    WITH CHECK (check_is_admin(auth.uid()));

-- Policy 2: Everyone can view available souvenirs with their tourist spot details
CREATE POLICY "public_view_souvenirs"
    ON public.souvenirs
    FOR SELECT
    TO anon, authenticated
    USING (
        tourist_spot_id IN (
            SELECT id FROM public.tourist_spots WHERE is_visible = true
        )
    );

-- Grant permissions
GRANT ALL ON public.souvenirs TO authenticated;
GRANT SELECT ON public.souvenirs TO anon;
GRANT ALL ON public.souvenirs TO service_role;


-- ============================================
-- HELPER FUNCTIONS (if not already exist)
-- ============================================

-- Ensure check_is_admin function exists
CREATE OR REPLACE FUNCTION check_is_admin(user_id UUID)
RETURNS BOOLEAN
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN
    RETURN EXISTS (
        SELECT 1 FROM public.profiles
        WHERE id = user_id AND user_type = 'admin'
    );
END;
$$;

-- Ensure update_updated_at_column function exists
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER
LANGUAGE plpgsql
AS $$
BEGIN
    NEW.updated_at = NOW();
    RETURN NEW;
END;
$$;
